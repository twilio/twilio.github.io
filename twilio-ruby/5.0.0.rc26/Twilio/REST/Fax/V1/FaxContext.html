def initialize(version, sid)
  super(version)

  # Path Solution
  @solution = {
      sid: sid,
  }
  @uri = "/Faxes/#{@solution[:sid]}"

  # Dependents
  @media = nil
end

def fetch
  params = Twilio::Values.of({})

  payload = @version.fetch(
      'GET',
      @uri,
      params,
  )

  FaxInstance.new(
      @version,
      payload,
      sid: @solution[:sid],
  )
end

def media(sid=:unset)
  raise ArgumentError, 'sid cannot be nil' if sid.nil?

  if sid != :unset
    return FaxMediaContext.new(
        @version,
        @solution[:sid],
        sid,
    )
  end

  unless @media
    @media = FaxMediaList.new(
        @version,
        fax_sid: @solution[:sid],
    )
  end

  @media
end

def update(status: :unset)
  data = Twilio::Values.of({
      'Status' => status,
  })

  payload = @version.update(
      'POST',
      @uri,
      data: data,
  )

  FaxInstance.new(
      @version,
      payload,
      sid: @solution[:sid],
  )
end
