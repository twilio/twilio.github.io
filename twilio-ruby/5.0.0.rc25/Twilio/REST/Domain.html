def request(method, uri, params={}, data={}, headers={}, auth=nil, timeout=nil)
  if uri.match(/^http/)
    url = uri
  else
    url = self.absolute_url(uri)
  end

  @client.request(
    @host,
    @port,
    method,
    url,
    params,
    data,
    headers,
    auth,
    timeout
  )
end
