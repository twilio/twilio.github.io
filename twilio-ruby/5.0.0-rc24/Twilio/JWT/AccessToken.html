  super(secret_key: secret,
        issuer: signing_key_sid,
        subject: account_sid,
        nbf: nbf,
        ttl: ttl,
        valid_until: valid_until)

  @signing_key_sid = signing_key_sid
  @secret = secret

  @nbf = nbf

end
